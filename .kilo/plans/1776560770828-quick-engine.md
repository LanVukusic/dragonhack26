```cpp
// Add these globals (around line 21)
uint8_t breathBrightness = 0;
bool breathRising = true;
uint8_t chaseIndex = 0;

// In loop(), replace effect routing:
void loop() {
    if(client.available()) {
        client.poll();
    }
    
    if (currentEffect == 0) {
        fill_solid(leds, NUM_LEDS, CRGB::Black);
    } else if (currentEffect == 1) {
        fill_solid(leds, NUM_LEDS, CRGB(currentR, currentG, currentB));
    } else if (currentEffect == 2) {
        // Breathing effect
        breathBrightness = beat8(60);  // Use FastLED's beat8 for smooth breathing
        CRGB color = CRGB(currentR, currentG, currentB);
        color *= breathBrightness;
        fill_solid(leds, NUM_LEDS, color);
    } else if (currentEffect == 3) {
        // Chase effect
        fill_solid(leds, NUM_LEDS, CRGB::Black);
        leds[chaseIndex] = CRGB(currentR, currentG, currentB);
        chaseIndex = (chaseIndex + 1) % NUM_LEDS;
    }

    FastLED.show();
    delay(20);
}
```
